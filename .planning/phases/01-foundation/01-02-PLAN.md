    ```typescript
    import i18n from 'i18next';
    import { initReactI18next } from 'react-i18next';
    import { getLocales } from 'expo-localization';
    import en from './en.json';
    import de from './de.json';

    const SUPPORTED_LANGS = ['en', 'de'] as const;
    type SupportedLang = (typeof SUPPORTED_LANGS)[number];

    function detectInitialLanguage(): SupportedLang {
      const locales = getLocales();
      const code = locales[0]?.languageCode ?? 'en';
      return (SUPPORTED_LANGS as readonly string[]).includes(code) ? (code as SupportedLang) : 'en';
    }

    void i18n
      .use(initReactI18next)
      .init({
        lng: detectInitialLanguage(),
        fallbackLng: 'en',
        supportedLngs: ['en', 'de'],
        resources: {
          en: { common: en.common, auth: en.auth, nav: en.nav },
          de: { common: de.common, auth: de.auth, nav: de.nav },
        },
        ns: ['common', 'auth', 'nav'],
        defaultNS: 'common',
        interpolation: {
          escapeValue: false, // React handles escaping
        },
        pluralSeparator: '_',
        compatibilityJSON: 'v4', // Phase 2+ plural keys work without re-init
        react: {
          useSuspense: false, // CRITICAL: React Native does not support React Suspense (RESEARCH Pitfall 5)
        },
      });

    export default i18n;
    ```

| Boundary | Description |
|----------|-------------|
| Device OS → App | Device locale read via `expo-localization`; trusted OS-provided value |
| Translation files → UI | Bundled JSON resources; no runtime fetch from untrusted source |

## STRIDE Threat Register (Plan 02 scope)

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-12 | T (Tampering) | Untranslated key fallback shown to user | mitigate | `fallbackLng: 'en'` + identical key sets in en.json/de.json (Task 1 verify) prevent users seeing raw key strings; Task 1 acceptance asserts recursive key parity |
| T-01-13 | I (Information Disclosure) | Auth error copy | mitigate | `auth.error_invalid_credentials` (single generic copy in both languages) and `auth.error_privacy_safe_reset` are present in both locales — prevents user-enumeration leaks (re-enforced in Plan 03) |
| T-01-14 | S (Spoofing) | Hydration mismatch on web rendering wrong-language text temporarily | mitigate | `useDidFinishSSR()` guard pattern: server renders English fallback, client hydrates with detected locale; Phase 1 smoke screen demonstrates the pattern, Plan 03 enforces it on every auth screen |
</threat_model>

<verification>
1. **Build verification:** `cd expo-app && npx expo start --web` boots without errors; smoke screen renders with `i18n active: <code>` line visible.
2. **Type verification:** `cd expo-app && npx tsc --noEmit` exits 0.
3. **Locale verification (manual):**
   - Browser with Accept-Language `en-US`: smoke screen shows `i18n active: en — Books`.
   - Browser with Accept-Language `de-DE` (or set OS to German + restart): smoke screen shows `i18n active: de — Bücher`.
4. **Key parity verification:** The Task 1 verify command runs a Node script that recursively compares en.json and de.json key sets — must report `OK <N> keys` and exit 0.
5. **Side-effect import verification:** Both `_layout.tsx` and `_layout.web.tsx` import `'../lib/i18n'`. Removing this import causes the smoke screen to render `i18n active: en — undefined` (or throw), proving the side-effect ordering matters.
</verification>
